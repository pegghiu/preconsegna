Timeline invio
Implementazione Supabase Edge Function

Crea questo file in supabase/functions/invia-fattura/index.ts: 

import { serve } from "https://deno.land/std/http/server.ts"

serve(async (req) => {
  // Leggi il payload dal frontend
  const { xml_base64, nome_file, progressivo } = await req.json()

  // Credenziali Aruba — da variabili d'ambiente Supabase
  const ARUBA_USER = Deno.env.get("ARUBA_USER")
  const ARUBA_PASS = Deno.env.get("ARUBA_PASS")
  const basicAuth = btoa(`${ARUBA_USER}:${ARUBA_PASS}`)

  // Invio a Aruba FatturaPA
  const res = await fetch(
    "https://fatturazioneelettronica.aruba.it/v1/fatture",
    {
      method: "POST",
      headers: {
        "Authorization": `Basic ${basicAuth}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify({ file: xml_base64, nome: nome_file })
    }
  )

  const data = await res.json()
  return new Response(JSON.stringify(data), {
    headers: { "Content-Type": "application/json" }
  })
})

Deploy: supabase functions deploy invia-fattura --no-verify-jwt
Imposta segreti: supabase secrets set ARUBA_USER=xxx ARUBA_PASS=yyy
